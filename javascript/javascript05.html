03. 비교 연산자

{
    let isEqual = (10 == 5); // 동등 비교
    let isStrictEqual = (10 === '10'); // 일치 비교
    let isNotEqual = (5 != 3); // 부등 비교
    let isNotStrictEqual = (5 !== '5'); // 부등 비교 (타입 비교)
    let isGreater = (10 > 5); // 크기 비교
    let isLessOrEqual = (3 <= 3); // 작거나 같음 비교                                                          
}

05. 비트 연산자

{
    let bitwiseAnd = 5 & 3; // 비트 AND
    let bitwiseOr = 5 | 3; // 비트 OR
    let bitwiseXor = 5 ^ 3; // 비트 XOR
    let bitwiseNot = ~5; // 비트 NOT
    let leftShift = 5 << 1; // 왼쪽 시프트
    let rightShift = 5 >> 1; // 오른쪽 시프트
    let unsignedRightShift = -5 >>> 1; // 부호 없는 오른쪽 시프트                                                                                               
}
