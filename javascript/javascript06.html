01. 문자열(String)

{
    let greeting = "Hello, World!";
    console.log(greeting); // 출력: Hello, World!                                    
}
결과 확인하기
Hello, World!

02. 숫자(Number)

{
    let number = 42;
    console.log(number); // 출력: 42                                                     
}
결과 확인하기
42

03. 불리언(Boolean)

{
    let isTrue = true;
    console.log(isTrue); // 출력: true                                                                           
}
결과 확인하기
true

04. null

{
    let nullValue = null;
    console.log(nullValue); // 출력: null                                                                                              
}
결과 확인하기
null

05. undefined

{
    let undefinedValue;
    console.log(undefinedValue); // 출력: undefined                                                                                                                
}
결과 확인하기
undefined

06. 심볼(Symbol)

{
    let key = Symbol('key');
    let obj = {};
    obj[key] = 'value';
    console.log(obj[key]); // 출력: value                                                                                                                                  
}
결과 확인하기
value

07. 객체(Object)

{
    let person = {
        name: 'John',
        age: 30,
        city: 'New York'
    };
    console.log(person.name); // 출력: John                                                                                                                                                    
}
결과 확인하기
John

08. 배열(Array)

{
    let numbers = [1, 2, 3, 4, 5];
    console.log(numbers.length); // 출력: 5                                                                                                                                                    
}
결과 확인하기
5

09. 함수(Function)

{
    function greet(name) {
        console.log("Hello, " + name + "!");
    }
    greet("Alice"); // 출력: Hello, Alice!                                                                                                                                                  
}
결과 확인하기
Hello, Alice!
